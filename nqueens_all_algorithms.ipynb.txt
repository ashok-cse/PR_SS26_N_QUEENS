    "## Shared bookkeeping: O(N) conflict counts\n",
    "\n",
    "We maintain three counters: `col[c]`, `d1[r+c]` for the \"/\" diagonals, and `d2[r-c+N-1]`\n",
    "for the \"\\\\\" diagonals. Total conflicts is `sum_k C(k, 2)` over each counter. Updates are\n",
    "O(1) per queen move using the identity `C(k+1,2) - C(k,2) = k` and `C(k-1,2) - C(k,2) = -(k-1)`.\n"
   ]
  },
  {
   "cell_type": "code",
   "metadata": {},

    "def init_counts(board, N):\n",
    "    \"\"\"Build column and diagonal count arrays in O(N).\"\"\"\n",
    "    col = [0] * N\n",
    "    d1 = [0] * (2 * N)\n",
    "    d2 = [0] * (2 * N)\n",
    "    for r in range(N):\n",
    "        c = board[r]\n",
    "        col[c] += 1\n",
    "        d1[r + c] += 1\n",
    "        d2[r - c + N - 1] += 1\n",
    "    return col, d1, d2\n",
    "\n",
    "\n",
    "def total_conflicts(col, d1, d2):\n",
    "    s = 0\n",
    "    for k in col:\n",
    "        if k > 1:\n",
    "            s += k * (k - 1) // 2\n",
    "    for k in d1:\n",
    "        if k > 1:\n",
    "            s += k * (k - 1) // 2\n",
    "    for k in d2:\n",
    "        if k > 1:\n",
    "            s += k * (k - 1) // 2\n",
    "    return s\n",
    "\n",
    "\n",
    "def total_conflicts_perm(d1, d2):\n",
    "    \"\"\"Permutation rep: only diagonal conflicts.\"\"\"\n",
    "    s = 0\n",
    "    for k in d1:\n",
    "        if k > 1:\n",
    "            s += k * (k - 1) // 2\n",
    "    for k in d2:\n",
    "        if k > 1:\n",
    "            s += k * (k - 1) // 2\n",
    "    return s\n",
    "\n",
    "\n",
    "def apply_swap(board, i, j, d1, d2, N):\n",
    "    \"\"\"Swap board[i] and board[j], update d1/d2 in place, return conflict delta.\"\"\"\n",
    "    ci, cj = board[i], board[j]\n",
    "    delta = 0\n",
    "    # Remove (i, ci)\n",
    "    delta -= d1[i + ci] - 1\n",
    "    d1[i + ci] -= 1\n",
    "    delta -= d2[i - ci + N - 1] - 1\n",
    "    d2[i - ci + N - 1] -= 1\n",
    "    # Remove (j, cj)\n",
    "    delta -= d1[j + cj] - 1\n",
    "    d1[j + cj] -= 1\n",
    "    delta -= d2[j - cj + N - 1] - 1\n",
    "    d2[j - cj + N - 1] -= 1\n",
    "    # Add (i, cj)\n",
    "    delta += d1[i + cj]\n",
    "    d1[i + cj] += 1\n",
    "    delta += d2[i - cj + N - 1]\n",
    "    d2[i - cj + N - 1] += 1\n",
    "    # Add (j, ci)\n",
    "    delta += d1[j + ci]\n",
    "    d1[j + ci] += 1\n",
    "    delta += d2[j - ci + N - 1]\n",
    "    d2[j - ci + N - 1] += 1\n",
    "    board[i], board[j] = cj, ci\n",
    "    return delta\n",
    "\n",
    "\n",
    "def swap_delta(board, i, j, d1, d2, N):\n",
    "    \"\"\"Return the conflict-delta a swap of (i, j) WOULD cause, without\n",
    "    leaving any side effect on board / d1 / d2. Uses apply_swap twice.\"\"\"\n",
    "    delta = apply_swap(board, i, j, d1, d2, N)\n",
    "    apply_swap(board, i, j, d1, d2, N)\n",
    "    return delta\n",
    "\n",
    "\n",
    "def min_conflicts_repair(board, N, steps=20):\n",
    "    \"\"\"Local search on a permutation. At each step pick a row that lies on\n",
    "    a conflicted diagonal and swap it with the row that most reduces total\n",
    "    diagonal conflicts. Mutates board in place; returns final total.\n",
    "\n",
    "    **Optimisation** \u2014 only OTHER conflicted rows are considered as swap\n",
    "    partners (with a single non-conflicted fallback). This drops each step\n",
    "    from O(N) to O(K), where K = number of currently conflicted rows. Once\n",
    "    the chromosome is mostly clean, K is tiny so the repair cost goes to\n",
    "    near-zero, which is what makes the memetic GA scale to N = 500.\n",
    "\n",
    "    Used after crossover and mutation in the hybrid GA and as a warm start\n",
    "    before SA at large N.\"\"\"\n",
    "    d1 = [0] * (2 * N)\n",
    "    d2 = [0] * (2 * N)\n",
    "    for r in range(N):\n",
    "        c = board[r]\n",
    "        d1[r + c] += 1\n",
    "        d2[r - c + N - 1] += 1\n",
    "    total = total_conflicts_perm(d1, d2)\n",
    "    for _ in range(steps):\n",
    "        if total == 0:\n",
    "            break\n",
    "        conflicted = [r for r in range(N)\n",
    "                      if d1[r + board[r]] > 1 or d2[r - board[r] + N - 1] > 1]\n",
    "        if not conflicted:\n",
    "            break\n",
    "        r1 = conflicted[0]\n",
    "        # Fast path: prefer swapping with OTHER conflicted rows (O(K)).\n",
    "        best_r2, best_delta = r1, 0\n",
    "        for r2 in conflicted:\n",
    "            if r2 == r1:\n",
    "                continue\n",
    "            delta = swap_delta(board, r1, r2, d1, d2, N)\n",
    "            if delta < best_delta:\n",
    "                best_delta = delta\n",
    "                best_r2 = r2\n",
    "        # Fallback: if no improvement among conflicted rows, scan ALL rows\n",
    "        # once. This is the O(N) cost \u2014 but it only triggers near\n",
    "        # convergence, when conflicts are sparse and an \"outside\" swap\n",
    "        # partner is the only way to clear them.\n",
    "        if best_r2 == r1:\n",
    "            conflicted_set = set(conflicted)\n",
    "            for r2 in range(N):\n",
    "                if r2 == r1 or r2 in conflicted_set:\n",
    "                    continue\n",
    "                delta = swap_delta(board, r1, r2, d1, d2, N)\n",
    "                if delta < best_delta:\n",
    "                    best_delta = delta\n",
    "                    best_r2 = r2\n",
    "        if best_r2 == r1:\n",
    "            break  # truly no improving swap available\n",
    "        apply_swap(board, r1, best_r2, d1, d2, N)\n",
    "        total += best_delta\n",
    "    return total\n"

    "## 1) Exhaustive DFS / Backtracking \u2014 with randomized restart\n",
    "\n",
    "Plain depth-first backtracking with `cols`, `d1`, `d2` sets is exact but its worst-case\n",
    "time grows factorially, and at `N=500` a single descent essentially never returns.\n",
    "\n",
    "The fix is a **Las-Vegas wrapper around the same backtracker**: at each row the column\n",
    "order is randomly shuffled, and a per-attempt backtrack budget is enforced. When the\n",
    "budget is exhausted the search abandons the current tree and restarts with a fresh\n",
    "random column order; the budget grows geometrically across restarts. The recursion\n",
    "itself is identical to plain backtracking \u2014 sets `cols`, `(r+c) in d1`, `(r-c) in d2`\n",
    "gate placements in O(1), and the first complete board (`r == N`) is returned.\n"
   ]
  },
  {
   "cell_type": "code",
   "metadata": {},
   "execution_count": null,
   "outputs": [],
   "source": [
    "class _DFSTimeout(Exception):\n",
    "    pass\n",
    "\n",
    "\n",
    "class _DFSRestart(Exception):\n",
    "    pass\n",
    "\n",
    "\n",
    "def dfs_solve(N, timeout, seed=SEED):\n",
    "    rng = random.Random(seed)\n",
    "    cols = set()\n",
    "    d1 = set()      # r + c\n",
    "    d2 = set()      # r - c\n",
    "    board = [-1] * N\n",
    "    backtracks = [0]   # mutable counters for closure\n",
    "    limit = [0]\n",
    "    start = time.time()\n",
    "\n",
    "    def backtrack(r):\n",
    "        if time.time() - start > timeout:\n",
    "            raise _DFSTimeout()\n",
    "        if backtracks[0] >= limit[0]:\n",
    "            raise _DFSRestart()\n",
    "        if r == N:\n",
    "            return True\n",
    "        order = list(range(N))\n",
    "        rng.shuffle(order)\n",
    "        for c in order:\n",
    "            if c in cols or (r + c) in d1 or (r - c) in d2:\n",
    "                continue\n",
    "            board[r] = c\n",
    "            cols.add(c); d1.add(r + c); d2.add(r - c)\n",
    "            if backtrack(r + 1):\n",
    "                return True\n",
    "            cols.discard(c); d1.discard(r + c); d2.discard(r - c)\n",
    "            board[r] = -1\n",
    "        backtracks[0] += 1\n",
    "        return False\n",
    "\n",
    "    old_limit = sys.getrecursionlimit()\n",
    "    sys.setrecursionlimit(max(old_limit, N + 1000))\n",
    "    base = max(N, 64)\n",
    "    attempt = 0\n",
    "    try:\n",
    "        while True:\n",
    "            if time.time() - start > timeout:\n",
    "                return None, \"timeout\"\n",
    "            attempt += 1\n",
    "            # geometric growth: 1, 1, 1, 2, 2, 2, 4, 4, 4, 8, ...\n",
    "            limit[0] = base * (1 << min((attempt - 1) // 3, 12))\n",
    "            backtracks[0] = 0\n",
    "            cols.clear(); d1.clear(); d2.clear()\n",
    "            board[:] = [-1] * N\n",
    "            try:\n",
    "                if backtrack(0):\n",
    "                    return board, 0\n",
    "            except _DFSRestart:\n",
    "                continue\n",
    "    except _DFSTimeout:\n",
    "        return None, \"timeout\"\n",
    "    except RecursionError:\n",
    "        return None, \"recursion_error\"\n",
    "    finally:\n",
    "        sys.setrecursionlimit(old_limit)\n"
   ]

    "## 2) Hill Climbing with Min-Conflicts\n",
    "\n",
    "For each step, pick a conflicted queen and move it to the column that minimises its conflicts.\n",
    "Ties broken at random. Random restarts whenever the search plateaus or exhausts a step budget.\n",
    "Conflict bookkeeping is incremental: O(N) per step for the column search, O(1) for accounting.\n"
   ]
  },
  {
   "cell_type": "code",
   "metadata": {},
   "execution_count": null,

    "def hill_climbing(N, timeout, seed=SEED):\n",
    "    rng = random.Random(seed)\n",
    "    start = time.time()\n",
    "    best_board = None\n",
    "    best_total = math.inf\n",
    "\n",
    "    def queen_attacks(r, c, col, d1, d2):\n",
    "        return (col[c] - 1) + (d1[r + c] - 1) + (d2[r - c + N - 1] - 1)\n",
    "\n",
    "    plateau_limit = max(30, N // 2)\n",
    "    max_steps = N * 20\n",
    "\n",
    "    while time.time() - start < timeout:\n",
    "        board = [rng.randrange(N) for _ in range(N)]\n",
    "        col, d1, d2 = init_counts(board, N)\n",
    "        total = total_conflicts(col, d1, d2)\n",
    "\n",
    "        if total < best_total:\n",
    "            best_total = total\n",
    "            best_board = board[:]\n",
    "        if total == 0:\n",
    "            return board, 0\n",
    "\n",
    "        plateau = 0\n",
    "        for _ in range(max_steps):\n",
    "            if time.time() - start > timeout:\n",
    "                return best_board, best_total\n",
    "\n",
    "            conflicted = [r for r in range(N)\n",
    "                          if queen_attacks(r, board[r], col, d1, d2) > 0]\n",
    "            if not conflicted:\n",
    "                break\n",
    "            r = rng.choice(conflicted)\n",
    "            old_c = board[r]\n",
    "\n",
    "            # Remove queen, track delta\n",
    "            delta = -(col[old_c] - 1) - (d1[r + old_c] - 1) - (d2[r - old_c + N - 1] - 1)\n",
    "            col[old_c] -= 1\n",
    "            d1[r + old_c] -= 1\n",
    "            d2[r - old_c + N - 1] -= 1\n",
    "\n",
    "            # Search best new column (O(N))\n",
    "            best_cost = None\n",
    "            best_cands = []\n",
    "            for c in range(N):\n",
    "                cost = col[c] + d1[r + c] + d2[r - c + N - 1]\n",
    "                if best_cost is None or cost < best_cost:\n",
    "                    best_cost = cost\n",
    "                    best_cands = [c]\n",
    "                elif cost == best_cost:\n",
    "                    best_cands.append(c)\n",
    "            new_c = rng.choice(best_cands)\n",
    "\n",
    "            delta += best_cost\n",
    "            col[new_c] += 1\n",
    "            d1[r + new_c] += 1\n",
    "            d2[r - new_c + N - 1] += 1\n",
    "            board[r] = new_c\n",
    "            new_total = total + delta\n",
    "\n",
    "            if new_total >= total:\n",
    "                plateau += 1\n",
    "            else:\n",
    "                plateau = 0\n",
    "            total = new_total\n",
    "\n",
    "            if total < best_total:\n",
    "                best_total = total\n",
    "                best_board = board[:]\n",
    "            if total == 0:\n",
    "                return board, 0\n",
    "            if plateau > plateau_limit:\n",
    "                break  # random restart\n",
    "\n",
    "    return best_board, best_total\n"

    "## 3) Simulated Annealing\n",
    "\n",
    "Permutation representation (board[r] is unique per r, so no column conflicts to track).\n",
    "Neighbour move = swap of two rows. `apply_swap` mutates the diagonal counts in place and\n",
    "returns the conflict delta in O(1), so each iteration is O(1). If the proposal is rejected,\n",
    "applying the swap again exactly undoes it and returns `-delta`. Temperature cools\n",
    "geometrically and reheats when it drops below `T_min`.\n",
    "\n",
    "**Paper-inspired warm start.** For `N >= WARM_START_N` the random permutation is first\n",
    "improved by `min_conflicts_repair`. This brings the initial conflict count from the\n",
    "typical random-permutation level (\u2248 a few \u00d7 N) down close to zero, so the Metropolis\n",
    "loop spends its iterations on the hard tail of the search rather than the easy bulk.\n"
   ]
  },
  {
   "cell_type": "code",
   "metadata": {},
   "execution_count": null,
   "outputs": [],
   "source": [
    "def simulated_annealing(N, timeout, seed=SEED, warm_start_n=WARM_START_N):\n",
    "    rng = random.Random(seed)\n",
    "    start = time.time()\n",
    "\n",
    "    board = list(range(N))\n",
    "    rng.shuffle(board)\n",
    "\n",
    "    # Optional min-conflicts warm start for large N.\n",
    "    if N >= warm_start_n:\n",
    "        warm_steps = min(N, 500)\n",
    "        min_conflicts_repair(board, N, steps=warm_steps)\n",
    "\n",
    "    d1 = [0] * (2 * N)\n",
    "    d2 = [0] * (2 * N)\n",
    "    for r in range(N):\n",
    "        c = board[r]\n",
    "        d1[r + c] += 1\n",
    "        d2[r - c + N - 1] += 1\n",
    "    total = total_conflicts_perm(d1, d2)\n",
    "\n",
    "    best_board = board[:]\n",
    "    best_total = total\n",
    "    if total == 0:\n",
    "        return board, 0\n",
    "\n",
    "    T = max(1.0, N / 2.0)\n",
    "    T_min = 1e-3\n",
    "    alpha = 0.9995\n",
    "    reheat_at = T_min\n",
    "    reheat_T = max(1.0, N / 4.0)\n",
    "\n",
    "    # Conflict-biased neighbour selection: paper-inspired tweak that makes\n",
    "    # SA tractable at large N once the warm start has pruned the bulk of the\n",
    "    # conflicts. We recompute the conflicted-row cache every \"refresh\"\n",
    "    # iterations because it goes stale as the board changes.\n",
    "    conflicted_cache = []\n",
    "    refresh = max(50, N // 4)\n",
    "    bias_prob = 0.85\n",
    "\n",
    "    max_iters = N * 5000\n",
    "    for it in range(max_iters):\n",
    "        if time.time() - start > timeout:\n",
    "            break\n",
    "        if total == 0:\n",
    "            break\n",
    "\n",
    "        if it % refresh == 0:\n",
    "            conflicted_cache = [\n",
    "                r for r in range(N)\n",
    "                if d1[r + board[r]] > 1 or d2[r - board[r] + N - 1] > 1\n",
    "            ]\n",
    "\n",
    "        if conflicted_cache and rng.random() < bias_prob:\n",
    "            i = rng.choice(conflicted_cache)\n",
    "        else:\n",
    "            i = rng.randrange(N)\n",
    "        j = rng.randrange(N)\n",
    "        if i == j:\n",
    "            continue\n",
    "\n",
    "        delta = apply_swap(board, i, j, d1, d2, N)\n",
    "        if delta <= 0 or rng.random() < math.exp(-delta / T):\n",
    "            total += delta\n",
    "            if total < best_total:\n",
    "                best_total = total\n",
    "                best_board = board[:]\n",
    "        else:\n",
    "            # Undo: same swap reverses it\n",
    "            apply_swap(board, i, j, d1, d2, N)\n",
    "\n",
    "        T *= alpha\n",
    "        if T < reheat_at:\n",
    "            T = reheat_T\n",
    "\n",
    "    return best_board, best_total\n"
   ]

    "## 4) Genetic Algorithm (memetic / hybrid)\n",
    "\n",
    "Permutation chromosomes, tournament selection, ordered crossover (OX), elitism. Lower\n",
    "fitness (diagonal-conflict count) is better.\n",
    "\n",
    "**Paper-inspired hybridisation.** The classical GA on N-Queens stalls a few queens\n",
    "short of the optimum because random swap mutation is blind to the structure of the\n",
    "conflicts. This implementation replaces plain mutation with a **conflict-aware\n",
    "mutation** \u2014 a row that lies on a conflicted diagonal is selected, and it is swapped\n",
    "with the partner that yields the largest local conflict reduction (random fallback to\n",
    "keep diversity). After both crossover and mutation, the child is passed through\n",
    "`min_conflicts_repair(child, steps=20)`, turning the algorithm into a memetic GA while\n",
    "preserving GA scaffolding (tournament selection, OX, elitism).\n"
   ]
  },
  {
   "cell_type": "code",
   "metadata": {},
   "execution_count": null,
   "outputs": [],
   "source": [
    "def genetic_algorithm(N, timeout, seed=SEED):\n",
    "    rng = random.Random(seed)\n",
    "    start = time.time()\n",
    "\n",
    "    POP = 120\n",
    "    ELITE = 10\n",
    "    TOURN = 5\n",
    "    MUT_RATE = 0.4\n",
    "    REPAIR_STEPS = 20\n",
    "\n",
    "    def fitness(board):\n",
    "        d1 = [0] * (2 * N)\n",
    "        d2 = [0] * (2 * N)\n",
    "        for r in range(N):\n",
    "            c = board[r]\n",
    "            d1[r + c] += 1\n",
    "            d2[r - c + N - 1] += 1\n",
    "        s = 0\n",
    "        for k in d1:\n",
    "            if k > 1:\n",
    "                s += k * (k - 1) // 2\n",
    "        for k in d2:\n",
    "            if k > 1:\n",
    "                s += k * (k - 1) // 2\n",
    "        return s\n",
    "\n",
    "    def make_individual():\n",
    "        b = list(range(N))\n",
    "        rng.shuffle(b)\n",
    "        return b\n",
    "\n",
    "    def tournament(pop_fit):\n",
    "        cands = rng.sample(pop_fit, TOURN)\n",
    "        return min(cands, key=lambda x: x[1])[0]\n",
    "\n",
    "    def ordered_crossover(p1, p2):\n",
    "        a, b = sorted(rng.sample(range(N), 2))\n",
    "        child = [-1] * N\n",
    "        child[a:b + 1] = p1[a:b + 1]\n",
    "        present = set(child[a:b + 1])\n",
    "        idx = (b + 1) % N\n",
    "        for gene in p2[b + 1:] + p2[:b + 1]:\n",
    "            if gene not in present:\n",
    "                child[idx] = gene\n",
    "                idx = (idx + 1) % N\n",
    "        return child\n",
    "\n",
    "    def conflict_aware_mutate(board):\n",
    "        \"\"\"Pick a row on a conflicted diagonal; swap it with the partner that\n",
    "        most reduces total diagonal conflicts (random fallback for diversity).\"\"\"\n",
    "        if rng.random() >= MUT_RATE:\n",
    "            return board\n",
    "        d1 = [0] * (2 * N)\n",
    "        d2 = [0] * (2 * N)\n",
    "        for r in range(N):\n",
    "            c = board[r]\n",
    "            d1[r + c] += 1\n",
    "            d2[r - c + N - 1] += 1\n",
    "        conflicted = [r for r in range(N)\n",
    "                      if d1[r + board[r]] > 1 or d2[r - board[r] + N - 1] > 1]\n",
    "        if not conflicted:\n",
    "            # No conflict to target \u2014 small random swap for diversity\n",
    "            i, j = rng.sample(range(N), 2)\n",
    "            board[i], board[j] = board[j], board[i]\n",
    "            return board\n",
    "        r1 = rng.choice(conflicted)\n",
    "        best_r2, best_delta = -1, 0\n",
    "        for r2 in range(N):\n",
    "            if r2 == r1:\n",
    "                continue\n",
    "            delta = swap_delta(board, r1, r2, d1, d2, N)\n",
    "            if delta < best_delta:\n",
    "                best_delta = delta\n",
    "                best_r2 = r2\n",
    "        if best_r2 < 0:\n",
    "            # No improving swap \u2014 random swap to escape local optimum\n",
    "            candidates = [r for r in range(N) if r != r1]\n",
    "            best_r2 = rng.choice(candidates)\n",
    "        board[r1], board[best_r2] = board[best_r2], board[r1]\n",
    "        return board\n",
    "\n",
    "    population = [make_individual() for _ in range(POP)]\n",
    "    pop_fit = [(ind, fitness(ind)) for ind in population]\n",
    "    pop_fit.sort(key=lambda x: x[1])\n",
    "    best_board = pop_fit[0][0][:]\n",
    "    best_fit = pop_fit[0][1]\n",
    "    if best_fit == 0:\n",
    "        return best_board, 0\n",
    "\n",
    "    while best_fit > 0:\n",
    "        if time.time() - start > timeout:\n",
    "            break\n",
    "        new_pop = [ind[:] for ind, _ in pop_fit[:ELITE]]\n",
    "        while len(new_pop) < POP:\n",
    "            if time.time() - start > timeout:\n",
    "                break\n",
    "            p1 = tournament(pop_fit)\n",
    "            p2 = tournament(pop_fit)\n",
    "            child = ordered_crossover(p1, p2)\n",
    "            min_conflicts_repair(child, N, steps=REPAIR_STEPS)\n",
    "            child = conflict_aware_mutate(child)\n",
    "            min_conflicts_repair(child, N, steps=REPAIR_STEPS)\n",
    "            new_pop.append(child)\n",
    "        pop_fit = [(ind, fitness(ind)) for ind in new_pop]\n",
    "        pop_fit.sort(key=lambda x: x[1])\n",
    "        if pop_fit[0][1] < best_fit:\n",
    "            best_fit = pop_fit[0][1]\n",
    "            best_board = pop_fit[0][0][:]\n",
    "            if best_fit == 0:\n",
    "                return best_board, 0\n",
    "\n",
    "    return best_board, best_fit\n"
